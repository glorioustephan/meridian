# Next.js App Router

This example demonstrates the complete Meridian workflow in a Next.js App Router project. It shows the file structure, a Server Component page that fetches data, a Meridian client component that handles interaction, and the precompile workflow that ties them together.

## Project structure

```
my-shop/
├── app/
│   ├── layout.tsx                    ← Root layout (Server Component)
│   └── products/
│       └── page.tsx                  ← Products page (Server Component)
├── src/
│   └── components/
```

## The Meridian source components

### AddToCartButton.tsx

A small interactive component for adding a product to the cart:

```tsx
// src/components/AddToCartButton.tsx
'use client';

import { Component, state, effect } from '@meridian/meridian';

interface AddToCartButtonProps {
  productId: string;
  productName: string;
  price: number;
  onAddToCart: (productId: string, quantity: number) => Promise<void>;
}

export class AddToCartButton extends Component<AddToCartButtonProps> {
  @state quantity = 1;
  @state status: 'idle' | 'adding' | 'added' | 'error' = 'idle';

  get label(): string {
    switch (this.status) {
      case 'adding': return 'Adding...';
      case 'added': return 'Added to cart!';
      case 'error': return 'Try again';
      default: return 'Add to cart';
    }
  }

  get isDisabled(): boolean {
    return this.status === 'adding' || this.status === 'added';
  }

  get total(): string {
    return (this.props.price * this.quantity).toFixed(2);
  }

  increment(): void {
    this.quantity = this.quantity + 1;
  }

  decrement(): void {
    if (this.quantity > 1) this.quantity = this.quantity - 1;
  }

  async handleAdd(): Promise<void> {
    this.status = 'adding';
    try {
      await this.props.onAddToCart(this.props.productId, this.quantity);
      this.status = 'added';
    } catch {
      this.status = 'error';
    }
  }

  @effect
  resetAfterAdd(): void {
    if (this.status === 'added') {
      const id = setTimeout(() => { this.status = 'idle'; }, 2000);
      // cleanup: nothing to do for the timeout — it runs once per 'added' state
    }
  }

  render() {
    return (
      <div className="add-to-cart">
        <div className="quantity-selector">
          <button onClick={this.decrement} disabled={this.quantity <= 1}>−</button>
          <span>{this.quantity}</span>
          <button onClick={this.increment}>+</button>
        </div>
        <p className="total">Total: ${this.total}</p>
        <button
          onClick={this.handleAdd}
          disabled={this.isDisabled}
          className={`add-button status-${this.status}`}
        >
          {this.label}
        </button>
      </div>
    );
  }
}
```

### ProductCard.tsx

A card component that displays product details and renders the `AddToCartButton`:

```tsx
// src/components/ProductCard.tsx
'use client';

import { Component, state } from '@meridian/meridian';
import { AddToCartButton } from '@meridian/components/AddToCartButton';

interface Product {
  id: string;
  name: string;
  description: string;
  price: number;
  imageUrl: string;
  inStock: boolean;
}

interface CartItem {
  productId: string;
  quantity: number;
}

interface ProductCardProps {
  product: Product;
}

export class ProductCard extends Component<ProductCardProps> {
  @state expanded = false;

  get truncatedDescription(): string {
    const desc = this.props.product.description;
    if (this.expanded || desc.length <= 120) return desc;
    return desc.slice(0, 120) + '…';
  }

  toggleExpanded(): void {
    this.expanded = !this.expanded;
  }

  async addToCart(productId: string, quantity: number): Promise<void> {
    const response = await fetch('/api/cart', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ productId, quantity }),
    });
    if (!response.ok) throw new Error('Failed to add to cart');
  }

  render() {
    const { product } = this.props;
    return (
      <article className="product-card">
        <img src={product.imageUrl} alt={product.name} />
        <div className="product-info">
          <h2>{product.name}</h2>
          <p>${product.price.toFixed(2)}</p>
          <p className="description">
            {this.truncatedDescription}
            {product.description.length > 120 && (
              <button onClick={this.toggleExpanded} className="read-more">
                {this.expanded ? 'Show less' : 'Read more'}
              </button>
            )}
          </p>
          {product.inStock ? (
            <AddToCartButton
              productId={product.id}
              productName={product.name}
              price={product.price}
              onAddToCart={this.addToCart}
            />
          ) : (
            <p className="out-of-stock">Out of stock</p>
          )}
        </div>
      </article>
    );
  }
}
```

## The Server Component page

```tsx
// app/products/page.tsx (Server Component — no 'use client')
import type { Metadata } from 'next';
import { ProductCard } from '@meridian/components/ProductCard';

export const metadata: Metadata = {
  title: 'Products',
  description: 'Browse our product catalog.',
};

interface Product {
  id: string;
  name: string;
  description: string;
  price: number;
  imageUrl: string;
  inStock: boolean;
}

async function getProducts(): Promise<Product[]> {
  const res = await fetch('https://api.myshop.com/products', {
    next: { revalidate: 300 }, // Revalidate every 5 minutes
  });
  if (!res.ok) throw new Error('Failed to fetch products');
  return res.json();
}

export default async function ProductsPage() {
  const products = await getProducts();

  return (
    <main className="products-page">
      <h1>Products</h1>
      <p>{products.length} items available</p>
      <div className="product-grid">
        {products.map(product => (
          <ProductCard key={product.id} product={product} />
        ))}
      </div>
    </main>
  );
}
```

The Server Component fetches the product list using `async/await` and Next.js's built-in fetch caching. It renders `ProductCard` as a child — the `'use client'` boundary is established by the generated `ProductCard.tsx` file. Next.js handles the boundary automatically.

## The precompile workflow end-to-end

```
Developer runs: pnpm dev

1. npm lifecycle runs predev script:
   meridian build

   - Scans src/components/AddToCartButton.tsx
   - Compiles → .meridian/generated/components/AddToCartButton.tsx

   - Scans src/components/ProductCard.tsx
   - Compiles → .meridian/generated/components/ProductCard.tsx

   Output: [meridian] built 2 files in 87ms

2. next dev starts

   - Reads app/products/page.tsx (Server Component)
   - Sees import from '@meridian/components/ProductCard'
   - Resolves via tsconfig paths alias to .meridian/generated/components/ProductCard.tsx
   - Processes ProductCard.tsx as a standard 'use client' React component
   - Processes AddToCartButton.tsx as a standard 'use client' React component
   - Builds and serves the application
```

The two phases are completely separate. Next.js never sees Meridian syntax — it only processes the generated React TSX files that were written to disk by `meridian build`.
